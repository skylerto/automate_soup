def setup(
  url: nil,
  username: nil,
  token: nil,
  password: nil
)
  @url = url
  @credentials = if token
                   token_credentials(username, token)
                 else
                   password_credentials(username, password)
                 end
  @api = AutomateSoup::API.new(self)
  self
end
